    <script src="https://cdn.tailwindcss.com"></script>

    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/three.js/0.160.0/three.min.js"></script>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/gsap/3.12.2/gsap.min.js"></script>

    

    <style>

        @import url('https://fonts.googleapis.com/css2?family=Dancing+Script:wght@700&display=swap');

        body { margin: 0; background: #050505; overflow: hidden; color: white; }

        canvas { display: block; }

        .glass {

            background: rgba(255, 255, 255, 0.05);

            backdrop-filter: blur(10px);

            border: 1px solid rgba(255, 255, 255, 0.1);

            box-shadow: 0 8px 32px 0 rgba(0, 0, 0, 0.37);

        .font-cursive { font-family: 'Dancing Script', cursive; }

        #video-container video { transform: scaleX(-1); }

<body>

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useRef, useEffect, useMemo, useLayoutEffect } = React;


        // --- 核心数学工具 ---

        const generateConePoints = (count, radius, height) => {

            const pts = [];

            for (let i = 0; i < count; i++) {

                const h = Math.random() * height;

                const r = (radius * (height - h)) / height;

                const theta = Math.random() * Math.PI * 2;

                pts.push(new THREE.Vector3(Math.cos(theta) * r, h - height / 2, Math.sin(theta) * r));

            }

            return pts;

        };


        const generateSpherePoints = (count, radius) => {

            const pts = [];

            for (let i = 0; i < count; i++) {

                const u = Math.random();

                const v = Math.random();

                const theta = 2 * Math.PI * u;

                const phi = Math.acos(2 * v - 1);

                pts.push(new THREE.Vector3(radius * Math.sin(phi) * Math.cos(theta), radius * Math.sin(phi) * Math.sin(theta), radius * Math.cos(phi)));

            }

            return pts;

        };


        // --- 3D 圣诞树组件 ---

        const ChristmasScene = ({ phase, setPhase }) => {

            const mountRef = useRef(null);

            const meshRef = useRef();

            const mouse = useRef(new THREE.Vector2());


            useEffect(() => {

                const scene = new THREE.Scene();

                const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);

                const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

                renderer.setSize(window.innerWidth, window.innerHeight);

                renderer.setPixelRatio(window.devicePixelRatio);

                mountRef.current.appendChild(renderer.domElement);


                // 灯光

                const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);

                scene.add(ambientLight);

                const pointLight = new THREE.PointLight(0xffd700, 2);

                pointLight.position.set(5, 5, 5);

                scene.add(pointLight);


                // 粒子系统

                const count = 5000;

                const geometry = new THREE.SphereGeometry(0.03, 8, 8);

                const material = new THREE.MeshStandardMaterial({ color: 0x2D5A27, emissive: 0x1a3316 });

                const mesh = new THREE.InstancedMesh(geometry, material, count);

                

                const treePoints = generateConePoints(count, 4, 8);

                const nebulaPoints = generateSpherePoints(count, 10);

                const currentPositions = treePoints.map(p => p.clone());


                const dummy = new THREE.Object3D();

                scene.add(mesh);

                meshRef.current = { mesh, treePoints, nebulaPoints, currentPositions };


                camera.position.z = 15;


                // 动画循环

                const animate = () => {

                    requestAnimationFrame(animate);

                    

                    // 简单的交互斥力

                    if (phase === 'tree') {

                        currentPositions.forEach((pos, i) => {

                            dummy.position.copy(pos);

                            dummy.updateMatrix();

                            mesh.setMatrixAt(i, dummy.matrix);

                        });

                        mesh.instanceMatrix.needsUpdate = true;

                    }

                    

                    renderer.render(scene, camera);

                };

                animate();


                // 响应式

                const handleResize = () => {

                    camera.aspect = window.innerWidth / window.innerHeight;

                    camera.updateProjectionMatrix();

                    renderer.setSize(window.innerWidth, window.innerHeight);

                };

                window.addEventListener('resize', handleResize);


                return () => {

                    window.removeEventListener('resize', handleResize);

                    mountRef.current?.removeChild(renderer.domElement);

                };

            }, []);


            // 监听阶段变化触发 GSAP

            useEffect(() => {

                if (phase === 'blooming' && meshRef.current) {

                    meshRef.current.currentPositions.forEach((pos, i) => {

                        gsap.to(pos, {

                            x: meshRef.current.nebulaPoints[i].x,

                            y: meshRef.current.nebulaPoints[i].y,

                            z: meshRef.current.nebulaPoints[i].z,

                            duration: 2,

                            ease: "expo.out",

                            onComplete: () => { if(i === 0) setPhase('nebula'); }

                        });

                    });

                } else if (phase === 'collapsing' && meshRef.current) {

                    meshRef.current.currentPositions.forEach((pos, i) => {

                        gsap.to(pos, {

                            x: meshRef.current.treePoints[i].x,

                            y: meshRef.current.treePoints[i].y,

                            z: meshRef.current.treePoints[i].z,

                            duration: 2,

                            ease: "power4.inOut",

                            onComplete: () => { if(i === 0) setPhase('tree'); }

                        });

                    });

                }

            }, [phase]);


            return <div ref={mountRef} className="fixed inset-0" />;

        };


        // --- 主应用 ---

        const App = () => {

            const [phase, setPhase] = useState('tree'); // tree, blooming, nebula, collapsing

            const [isCamOpen, setIsCamOpen] = useState(false);

            const [isPlaying, setIsPlaying] = useState(false);

            const audioRef = useRef(new Audio('https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3')); // 示例BGM


            const togglePlay = () => {

                if (isPlaying) audioRef.current.pause();

                else audioRef.current.play();

                setIsPlaying(!isPlaying);

            };


            return (

                <div className="relative w-full h-screen">

                    <ChristmasScene phase={phase} setPhase={setPhase} />


                    {/* UI Overlay */}

                    <div className="relative z-10 pointer-events-none flex flex-col justify-between h-full p-8 text-white">

                        <div className="flex justify-between items-start pointer-events-auto">

                            <div className="glass p-6 rounded-2xl border-white/20">

                                <h2 className="text-xs tracking-widest opacity-60 uppercase mb-2">Current Phase</h2>

                                <p className="text-2xl font-bold tracking-tighter capitalize">{phase}</p>

                            </div>


                            <div id="video-container" className="glass w-48 h-36 rounded-2xl overflow-hidden relative border-white/20">

                                {!isCamOpen ? (

                                    <div className="w-full h-full flex items-center justify-center text-[10px] opacity-40 text-center px-4">

                                        CAMERA OFF<br/>CLICK "ENABLE GESTURE"

                                    </div>

                                ) : (

                                    <div className="w-full h-full bg-black/40 animate-pulse flex items-center justify-center text-xs">Loading AI Model...</div>

                                )}

                            </div>

                        </div>


                        <div className="text-center">

                            <h1 className="text-7xl md:text-9xl font-cursive mb-4 pointer-events-none select-none drop-shadow-2xl text-yellow-500/80">

                                Merry Christmas

                            </h1>

                        </div>


                        <div className="flex flex-col items-center gap-6 pointer-events-auto">

                            <div className="flex gap-4">

                                <button 

                                    onClick={() => setPhase(phase === 'tree' ? 'blooming' : 'collapsing')}

                                    className="px-8 py-3 glass rounded-full hover:bg-white/10 transition-all font-bold tracking-widest text-xs uppercase"

                                >

                                    {phase === 'tree' ? 'Trigger Bloom (Open Palm)' : 'Reset Tree (Close Fist)'}

                                </button>

                                <button 

                                    onClick={() => setIsCamOpen(!isCamOpen)}

                                    className="px-8 py-3 glass rounded-full hover:bg-white/10 transition-all font-bold tracking-widest text-xs uppercase"

                                >

                                    {isCamOpen ? 'Disable Gesture' : 'Enable Gesture'}

                                </button>

                            </div>


                            <div className="glass px-8 py-4 rounded-full flex items-center gap-4">

                                <div className={`w-8 h-8 rounded-full border-2 border-white/30 flex items-center justify-center ${isPlaying ? 'animate-spin' : ''}`}>

                                    ❄️

                                </div>

                                <div className="text-left leading-none">

                                    <p className="text-[10px] opacity-50 uppercase tracking-tighter">Now Playing</p>

                                    <p className="text-sm font-semibold">Merry Christmas Mr. Lawrence</p>

                                </div>

                                <button onClick={togglePlay} className="ml-4 hover:scale-110 transition-transform">

                                    {isPlaying ? '⏸️' : '▶️'}

                                </button>

                            </div>

                        </div>

                    </div>


                    {/* Background Noise/Static */}

                    <div className="fixed inset-0 pointer-events-none opacity-[0.03] bg-[url('https://grainy-gradients.vercel.app/noise.svg')]"></div>

                </div>

            );

        };


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>

</body>
